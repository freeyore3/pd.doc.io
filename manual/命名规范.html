常量

  • 使用首字母大写

变量

  • 局部变量:首字母小写。
  • 私有与保护成员变量:_首字母小写
  • 公有成员变量:首字母小写

namespace Framework.Style 
{	
	using UnityEngine;
	using UnityEngine.UI;
	using System.Collections;
	using System.Collections.Generic;
	
	// 展示编码风格
	public enum EnumStyle
	{
		Style1 = 0,
		Style2,
		Style3,
		MaxValue
	}

    struct StructStyle
    {
        public int index;
        public string str;
        public Vector3 pos;
        public Quaternion rotation;
    }

	public class ProgrammingStyle : MonoBehaviour 
	{
		// 私有成员变量
		Button _btn;
		int _curSelectIndex;
		Vector3[] _positions;
		GameObject _view;
		EnumStyle _enumStyle;
		StructStyle _struct;

		// public类型变量首字母小写
		public int lastIndex = 0;

		//public只读变量首字母小写
		public readonly int initIndex;

		// 常量使用首字母大写
		public const int First = 1;

		//属性使用首字母大写
		public int CurSelectIndex 
		{
			get { return _curSelectIndex; }
		}

		// 函数
		void Start () 
		{
			_view = transform.Find ("MainView").GetComponent<Button>();

			// 临时变量命名采用首字母小写驼峰式
			GameObject firstPosGo = transform.Find ("FirstPosGo").gameObject;
		}
	}
}
